36 // a uniform serialization mechanism for both object and fundamental
37 // types. For each serializable class simply implement a member method called
38 // serialize that takes no arguments, and create an in class typedef
39 // that is equal to the return type of this method.
40 template<class T>
41 typename T::SerializedDataType serialize(T& obj){
42  LOGWireFormat("serialize %s", typeid(T).name());
43  return obj.serialize();
44 }
