143 constexpr bool mustBeEncryptedRx(UartOpcodeRx opCode) {
144  switch (opCode) {
145  case UartOpcodeRx::UART_OPCODE_RX_HELLO:
146  case UartOpcodeRx::UART_OPCODE_RX_SESSION_NONCE:
147  case UartOpcodeRx::UART_OPCODE_RX_HEARTBEAT: // optional
148  case UartOpcodeRx::UART_OPCODE_RX_STATUS: // optional
149  case UartOpcodeRx::UART_OPCODE_RX_GET_MAC:
150  case UartOpcodeRx::UART_OPCODE_RX_HUB_DATA_REPLY: // optional
151  return false;
152  default:
153  if (opCode >= 50000) {
154  return false;
155  }
156  return true;
157  }
