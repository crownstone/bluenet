113 template<class T>
114 StreamBufferAccessor& get(T* &packet_ptr){
115 if( !status_check() ){
116 // failed status check
117 // ignore get request when failure
118 return *this;
119 }
120
121 if( !size_check<T>() ){
122 // failed size check
123 packet_ptr = nullptr;
124 internal_status = ERR_NO_SPACE;
125 } else {
126 // write to the out-parameter packet_ptr
127 packet_ptr = reinterpret_cast<T*>(buff_curr);
128 increment_buff<T>();
129 }
130
131 return *this;
132 }
133
140 template<class T>
141 T get(){
142 if( !size_check<T>() && !status_check() ){
143 increment_buff<T>();
144 return *reinterpret_cast<T*>(buff_curr);
145 }
146
147 // try to return a default when possible
148 internal_status = ERR_NO_SPACE;
149 return T();
150 }

158 template<class T>
159 void put(T packet){
160 if( size_check<T>() && status_check() ){
161 // copy into the buffer, after that, increment
162 *reinterpret_cast<T*>(buff_curr) = packet;
163 increment_buff<T>();
164 } else {
165 // failed size_check
166 internal_status = ERR_NO_SPACE;
167 }
168 }
169
170protected:
171 // returns false on failure
172 template<class T>
173 inline bool size_check(){
174 return buff_curr + sizeof(T) <= buff_end;
175 }
176
177 template<class T>
178 inline void increment_buff(){
179 buff_curr += sizeof(T);
180 }
181
182 // returns false on failure
183 inline bool status_check(){
184 return internal_status == ERR_SUCCESS;
185 }
