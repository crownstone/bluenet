109  constexpr uint32_t fileNameHash(const char* str, size_t strLen) {
110  uint32_t hash = 5381;
111  for (size_t i = strLen-1; i >= 0; --i) {
112  if (str[i] == '/') {
113  return hash;
114  }
115  hash = ((hash << 5) + hash) + str[i]; // hash * 33 + c
116  }
117  return hash;
118  }

197  // Uses the fold expression, a handy way to replace a recursive call.
198  template<class... Args>
199  void cs_log_args(uint32_t fileNameHash, uint32_t lineNumber, uint8_t logLevel, bool addNewLine, const Args&... args) {
200  uart_msg_log_header_t header;
201  header.header.fileNameHash = fileNameHash;
202  header.header.lineNumber = lineNumber;
203  header.header.logLevel = logLevel;
204  header.header.flags.newLine = addNewLine;
205  header.numArgs = 0;
206 
207  // Get number of arguments and size of all arguments.
208  size_t totalSize = sizeof(header);
209  (cs_log_add_arg_size(totalSize, header.numArgs, args), ...);
210 
211  // Write the header.
212  cs_log_start(totalSize, header);
213 
214  // Write each argument.
215  (cs_log_arg(args), ...);
216 
217  // Finalize the uart msg.
218  cs_log_end();
219  }
