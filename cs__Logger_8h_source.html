220constexpr uint32_t fileNameHash(const char* str, size_t strLen) {
221 uint32_t hash = 5381;
222 for (int32_t i = strLen - 1; i >= 0; --i) {
223 if (str[i] == '/') {
224 return hash;
225 }
226 hash = ((hash << 5) + hash) + str[i]; // hash * 33 + c
227 }
228 return hash;
229}

571// Uses the fold expression, a handy way to replace a recursive call.
572template <class... Args>
573void cs_log_args_no_fmt(
574 uint32_t fileNameHash, uint32_t lineNumber, uint8_t logLevel, bool addNewLine, const Args&... args) {
575 uart_msg_log_header_t header;
576 header.header.fileNameHash = fileNameHash;
577 header.header.lineNumber = lineNumber;
578 header.header.logLevel = logLevel;
579 header.header.flags.newLine = addNewLine;
580 header.numArgs = 0;
581
582 // Get number of arguments and size of all arguments.
583 size_t totalSize = sizeof(header);
584 (cs_log_add_arg_size(totalSize, header.numArgs, args), ...);
585
586 // Write the header.
587 cs_log_start(totalSize, header);
588
589 // Write each argument.
590 (cs_log_arg(args), ...);
591
592 // Finalize the uart msg.
593 cs_log_end();
594}
