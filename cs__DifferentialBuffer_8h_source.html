67  uint16_t getMaxByteSize(uint16_t capacity) { return 2 + 2*sizeof(T) + capacity-1; }
68  uint16_t getMaxByteSize() { return getMaxByteSize(_capacity); }
69  uint16_t getMaxSize(uint16_t byteSize) { return byteSize - 2 - 2*sizeof(T) + 1; }

71  bool init() {
72  if (_buffer != NULL) {
73  return false;
74  }
75  // Allocate memory
76  _buffer = (differential_buffer_t<T>*)calloc(getMaxByteSize(), sizeof(uint8_t));
77  if (_buffer == NULL) {
78  LOGw("Could not allocate memory");
79  return false;
80  }
81 // LOGd("Allocated memory at %u", _buffer);
82  _allocatedSelf = true;
83  // Also call clear to make sure we start with a clean buffer
84  clear();
85  return true;
86  }
87 
88  bool deinit() {
89  if (_buffer != NULL && _allocatedSelf) {
90  free(_buffer);
91  }
92  _allocatedSelf = false;
93  _buffer = NULL;
94  return true;
95  }
96 
97  bool assign(buffer_ptr_t buffer, uint16_t bufferSize) {
98  if (getMaxSize(bufferSize) < _capacity || _allocatedSelf) {
99  LOGd("Could not assign at %u", buffer);
100  LOGd("%u < %u", getMaxSize(bufferSize), _capacity);
101  return false;
102  }
103 // LOGd("assign at %u", buffer);
104  _buffer = (differential_buffer_t<T>*) buffer;
105  // Also call clear to make sure we start with a clean buffer
106  clear();
107  return true;
108  }
109 
110  bool release() {
111  if (_allocatedSelf) {
112  return false;
113  }
114  _buffer = NULL;
115  return true;
116  }

143  bool push(T value) {
144  if (full()) {
145  return false;
146  }
147 // LOGd("push %u at %u buffer=%u", value, _buffer->length, _buffer);
148  if (empty()) {
149  _buffer->firstVal = value;
150  _buffer->lastVal = value;
151  _buffer->length = 1;
152  return true;
153  }
154 
155  int32_t diff = (int32_t)value - _buffer->lastVal;
156 // LOGd("diff=%i", diff);
157  if (diff > 127 || diff < -127) {
158  LOGw("diff too large! %u - %u", value, _buffer->lastVal);
159  clear();
160  return false;
161  }
162  _buffer->array[_buffer->length - 1] = diff;
163  _buffer->length++;
164  _buffer->lastVal = value;
165  return true;
166  }
167 
168  T pop() {
169  assert(!empty(), "Buffer is empty");
170  T value = _buffer->lastVal;
171  if (size() == 1) {
172  clear();
173  }
174  else {
175  _buffer->lastVal -= _buffer->array[_buffer->length-- - 1];
176  }
177  return value;
178  }
179 
180  T peekBack() const {
181  assert(!empty(), "Buffer is empty");
182  return _buffer->lastVal;
183  }
184 
185  T peekFront() const {
186  assert(!empty(), "Buffer is empty");
187  return _buffer->firstVal;
188  }
189 
190  bool getValue(T& value, const uint16_t index) const {
191  if (index >= size()) {
192  return false;
193  }
194  if (index == 0) {
195 // LOGd("getVal %u", index);
196  value = _buffer->firstVal;
197  return true;
198  }
199 // LOGd("getVal %u diff=%i buffer=%u array=%u", index, _buffer->array[index - 1], _buffer, _buffer->array);
200  value += _buffer->array[index - 1];
201  return true;
202  }
