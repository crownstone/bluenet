108 template<typename T, typename Allocator>
109 struct is_inplace_allocated
110 {
111 static const bool value
112 // so that it fits
113 = sizeof(T) <= sizeof(functor_padding)
114 // so that it will be aligned
115 && std::alignment_of<functor_padding>::value % std::alignment_of<T>::value == 0
116 // so that we can offer noexcept move
117 && std::is_nothrow_move_constructible<T>::value
118 // so that the user can override it
119 && !force_function_heap_allocation<T>::value;
120 };

212template<typename T, typename Allocator, typename Enable = void>
213struct function_manager_inplace_specialization
214{
215 template<typename Result, typename... Arguments>
216 static Result call(const functor_padding & storage, Arguments... arguments)
217 {
218 // do not call get_functor_ref because I want this function to be fast
219 // in debug when nothing gets inlined
220 return const_cast<T &>(reinterpret_cast<const T &>(storage))(FUNC_FORWARD(Arguments, arguments)...);
221 }
222
223 static void store_functor(manager_storage_type & storage, T to_store)
224 {
225 new (&get_functor_ref(storage)) T(FUNC_FORWARD(T, to_store));
226 }
227 static void move_functor(manager_storage_type & lhs, manager_storage_type && rhs) FUNC_NOEXCEPT
228 {
229 new (&get_functor_ref(lhs)) T(FUNC_MOVE(get_functor_ref(rhs)));
230 }
231 static void destroy_functor(Allocator &, manager_storage_type & storage) FUNC_NOEXCEPT
232 {
233 get_functor_ref(storage).~T();
234 }
235 static T & get_functor_ref(const manager_storage_type & storage) FUNC_NOEXCEPT
236 {
237 return const_cast<T &>(reinterpret_cast<const T &>(storage.functor));
238 }
239};
240template<typename T, typename Allocator>
241struct function_manager_inplace_specialization<T, Allocator, typename std::enable_if<!is_inplace_allocated<T, Allocator>::value>::type>
242{
243 template<typename Result, typename... Arguments>
244 static Result call(const functor_padding & storage, Arguments... arguments)
245 {
246 // do not call get_functor_ptr_ref because I want this function to be fast
247 // in debug when nothing gets inlined
248 return (*reinterpret_cast<const typename std::allocator_traits<Allocator>::pointer &>(storage))(FUNC_FORWARD(Arguments, arguments)...);
249 }
250
251 static void store_functor(manager_storage_type & self, T to_store)
252 {
253 Allocator & allocator = self.get_allocator<Allocator>();;
254 static_assert(sizeof(typename std::allocator_traits<Allocator>::pointer) <= sizeof(self.functor), "The allocator's pointer type is too big");
255 typename std::allocator_traits<Allocator>::pointer * ptr = new (&get_functor_ptr_ref(self)) typename std::allocator_traits<Allocator>::pointer(std::allocator_traits<Allocator>::allocate(allocator, 1));
256 std::allocator_traits<Allocator>::construct(allocator, *ptr, FUNC_FORWARD(T, to_store));
257 }
258 static void move_functor(manager_storage_type & lhs, manager_storage_type && rhs) FUNC_NOEXCEPT
259 {
260 static_assert(std::is_nothrow_move_constructible<typename std::allocator_traits<Allocator>::pointer>::value, "we can't offer a noexcept swap if the pointer type is not nothrow move constructible");
261 new (&get_functor_ptr_ref(lhs)) typename std::allocator_traits<Allocator>::pointer(FUNC_MOVE(get_functor_ptr_ref(rhs)));
262 // this next assignment makes the destroy function easier
263 get_functor_ptr_ref(rhs) = nullptr;
264 }
265 static void destroy_functor(Allocator & allocator, manager_storage_type & storage) FUNC_NOEXCEPT
266 {
267 typename std::allocator_traits<Allocator>::pointer & pointer = get_functor_ptr_ref(storage);
268 if (!pointer) return;
269 std::allocator_traits<Allocator>::destroy(allocator, pointer);
270 std::allocator_traits<Allocator>::deallocate(allocator, pointer, 1);
271 }
272 static T & get_functor_ref(const manager_storage_type & storage) FUNC_NOEXCEPT
273 {
274 return *get_functor_ptr_ref(storage);
275 }
276 static typename std::allocator_traits<Allocator>::pointer & get_functor_ptr_ref(manager_storage_type & storage) FUNC_NOEXCEPT
277 {
278 return reinterpret_cast<typename std::allocator_traits<Allocator>::pointer &>(storage.functor);
279 }
280 static const typename std::allocator_traits<Allocator>::pointer & get_functor_ptr_ref(const manager_storage_type & storage) FUNC_NOEXCEPT
281 {
282 return reinterpret_cast<const typename std::allocator_traits<Allocator>::pointer &>(storage.functor);
283 }
284};

332 template<typename T, typename Allocator>
333 static void templated_call_move_and_destroy(manager_storage_type & lhs, manager_storage_type && rhs)
334 {
335 typedef function_manager_inplace_specialization<T, Allocator> specialization;
336 specialization::move_functor(lhs, FUNC_MOVE(rhs));
337 specialization::destroy_functor(rhs.get_allocator<Allocator>(), rhs);
338 create_manager<T, Allocator>(lhs, FUNC_MOVE(rhs.get_allocator<Allocator>()));
339 rhs.get_allocator<Allocator>().~Allocator();
340 }
341 template<typename T, typename Allocator>
342 static void templated_call_copy(manager_storage_type & lhs, const manager_storage_type & rhs)
343 {
344 typedef function_manager_inplace_specialization<T, Allocator> specialization;
345 create_manager<T, Allocator>(lhs, Allocator(rhs.get_allocator<Allocator>()));
346 specialization::store_functor(lhs, specialization::get_functor_ref(rhs));
347 }
348 template<typename T, typename Allocator>
349 static void templated_call_destroy(manager_storage_type & self)
350 {
351 typedef function_manager_inplace_specialization<T, Allocator> specialization;
352 specialization::destroy_functor(self.get_allocator<Allocator>(), self);
353 self.get_allocator<Allocator>().~Allocator();
354 }
355 template<typename T, typename Allocator>
356 static void templated_call_copy_functor_only(manager_storage_type & lhs, const manager_storage_type & rhs)
357 {
358 typedef function_manager_inplace_specialization<T, Allocator> specialization;
359 specialization::store_functor(lhs, specialization::get_functor_ref(rhs));
360 }
361# ifndef FUNC_NO_RTTI
362 template<typename T, typename>
363 static const std::type_info & templated_call_type_id()
364 {
365 return typeid(T);
366 }
367 template<typename T, typename Allocator>
368 static void * templated_call_target(const manager_storage_type & self, const std::type_info & type)
369 {
370 typedef function_manager_inplace_specialization<T, Allocator> specialization;
371 if (type == typeid(T))
372 return &specialization::get_functor_ref(self);
373 else
374 return nullptr;
375 }
376# endif

405template<typename Result, typename... Arguments>
406class function<Result (Arguments...)>
407 : public detail::typedeffer<Result, Arguments...>
408{
409public:
410 function() FUNC_NOEXCEPT
411 {
412 initialize_empty();
413 }
414 function(std::nullptr_t) FUNC_NOEXCEPT
415 {
416 initialize_empty();
417 }
418 function(function && other) FUNC_NOEXCEPT
419 {
420 initialize_empty();
421 swap(other);
422 }
423 function(const function & other)
424 : call(other.call)

427 }
428 template<typename T>
429 function(T functor,
430 typename std::enable_if<detail::is_valid_function_argument<T, Result (Arguments...)>::value, detail::empty_struct>::type = detail::empty_struct()) FUNC_TEMPLATE_NOEXCEPT(T, std::allocator<typename detail::functor_type<T>::type>)
431 {
432 if (detail::is_null(functor))
433 {
434 initialize_empty();
435 }
436 else
437 {
438 typedef typename detail::functor_type<T>::type functor_type;
439 initialize(detail::to_functor(FUNC_FORWARD(T, functor)), std::allocator<functor_type>());
440 }
441 }
442 template<typename Allocator>
443 function(std::allocator_arg_t, const Allocator &)
444 {
445 // ignore the allocator because I don't allocate
446 initialize_empty();
447 }
448 template<typename Allocator>
449 function(std::allocator_arg_t, const Allocator &, std::nullptr_t)
450 {
451 // ignore the allocator because I don't allocate
452 initialize_empty();
453 }
454 template<typename Allocator, typename T>
455 function(std::allocator_arg_t, const Allocator & allocator, T functor,
456 typename std::enable_if<detail::is_valid_function_argument<T, Result (Arguments...)>::value, detail::empty_struct>::type = detail::empty_struct())
457 FUNC_TEMPLATE_NOEXCEPT(T, Allocator)
458 {
459 if (detail::is_null(functor))
460 {
461 initialize_empty();
462 }
463 else
464 {
465 initialize(detail::to_functor(FUNC_FORWARD(T, functor)), Allocator(allocator));
466 }

468 template<typename Allocator>
469 function(std::allocator_arg_t, const Allocator & allocator, const function & other)
470 : call(other.call)
471 {
472 typedef typename std::allocator_traits<Allocator>::template rebind_alloc<function> MyAllocator;
473
474 // first try to see if the allocator matches the target type
475 detail::manager_type manager_for_allocator = &detail::get_default_manager<typename std::allocator_traits<Allocator>::value_type, Allocator>();
476 if (other.manager_storage.manager == manager_for_allocator)
477 {
478 detail::create_manager<typename std::allocator_traits<Allocator>::value_type, Allocator>(manager_storage, Allocator(allocator));
479 manager_for_allocator->call_copy_functor_only(manager_storage, other.manager_storage);
480 }
481 // if it does not, try to see if the target contains my type. this
482 // breaks the recursion of the last case. otherwise repeated copies
483 // would allocate more and more memory
484 else
485 {
486 detail::manager_type manager_for_function = &detail::get_default_manager<function, MyAllocator>();
487 if (other.manager_storage.manager == manager_for_function)
488 {
489 detail::create_manager<function, MyAllocator>(manager_storage, MyAllocator(allocator));
490 manager_for_function->call_copy_functor_only(manager_storage, other.manager_storage);
491 }
492 else
493 {
494 // else store the other function as my target
495 initialize(other, MyAllocator(allocator));
496 }
497 }
498 }

528 void swap(function & other) FUNC_NOEXCEPT
529 {
530 detail::manager_storage_type temp_storage;
531 other.manager_storage.manager->call_move_and_destroy(temp_storage, FUNC_MOVE(other.manager_storage));
532 manager_storage.manager->call_move_and_destroy(other.manager_storage, FUNC_MOVE(manager_storage));
533 temp_storage.manager->call_move_and_destroy(manager_storage, FUNC_MOVE(temp_storage));
534
535 std::swap(call, other.call);
536 }
