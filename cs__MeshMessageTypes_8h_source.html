180 inline void push_state_item(state_message_t* message, state_item_t* item) {
181  if (++message->size > MAX_STATE_ITEMS) {
182  message->head = (message->head + 1) % MAX_STATE_ITEMS;
183  --message->size;

189 inline bool peek_next_state_item(state_message_t* message, state_item_t** item, int16_t& index) {
190  if (message->size == 0) {
191  return false;
192  } else if (index == -1) {
193  index = message->head;
194  } else {
