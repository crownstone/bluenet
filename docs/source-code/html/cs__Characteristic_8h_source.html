497  void written(uint16_t len) {
499  //TODO: have this from the stack directly.
500  if (EncryptionHandler::getInstance().allowedToWrite()) {
501  LOGi("Not allowed to write, disconnect in progress")
502  return;
503  }
504 
505  setGattValueLength(len);
506 
507  EncryptionAccessLevel accessLevel = NOT_SET;
508  // when using encryption, the packet needs to be decrypted
509  if (_status.aesEncrypted && _minAccessLevel < ENCRYPTION_DISABLED) {
510 
511  // the arithmetic type- and the string type characteristics have their hardcoded length in the valueLength
512  // the getValueLength() for those two types will always be the same and the setValueLength does nothing there.
513  // In the case of a characteristic with a dynamic buffer length we need to set the length ourselves.
514  // To do this we assume the length of the data is the same as the encrypted buffer minus the overhead for the encryption.
515  // The result can be zero padded but generally the payload has it's own length indication.
516  uint16_t decryptionBufferLength = EncryptionHandler::calculateDecryptionBufferLength(getGattValueLength());
517  setValueLength(decryptionBufferLength);
518 
519  bool success = EncryptionHandler::getInstance().decrypt(
520  getGattValuePtr(),
521  getGattValueLength(),
522  getValuePtr(),
523  getValueLength(),
524  accessLevel
525  );
526 
527  // disconnect on failure or if the user is not authenticated
528  if (!success || !EncryptionHandler::getInstance().allowAccess(_minAccessLevel , accessLevel)) {
529  LOGi("insufficient access")
530  EncryptionHandler::getInstance().closeConnectionAuthenticationFailure();
531  return;
532  }
533  }
534  else {
535  accessLevel = ENCRYPTION_DISABLED;
536  setValueLength(len);
537  }
538 
539 
540 
541  LOGd("%s: onWrite()", _name);
542 
543  _callbackOnWrite(accessLevel, getValue());
544  }
